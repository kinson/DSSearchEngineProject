4  algorithm presented in
5 
6  Porter, 1980, An algorithm for suffix stripping, Program, Vol. 14,
7  no. 3, pp 130-137,
8 
9  only differing from it at the points marked --DEPARTURE-- below.

18

20  'stem' takes a list of inputs and sends the stemmed equivalent to
21  stdout.

49 /* The main part of the stemming algorithm starts here. b is a buffer
50  holding a word to be stemmed. The letters are in b[k0], b[k0+1] ...
51  ending at b[k]. In fact k0 = 0 in this demo program. k is readjusted
52  downwards as the stemming progresses. Zero termination is not in fact
53  used in the algorithm.
54 
55  Note that only lower case sequences are stemmed. Forcing to lower case
56  should be done before stem(...) is called.
57 */
58 
59 static char * b; /* buffer for word to be stemmed */
60 static int k,k0,j; /* j is a general offset into the string */

62 /* cons(i) is TRUE <=> b[i] is a consonant. */
63 
64 static int cons(int i)
65 { switch (b[i])
66  { case 'a': case 'e': case 'i': case 'o': case 'u': return FALSE;
67  case 'y': return (i==k0) ? TRUE : !cons(i-1);
68  default: return TRUE;
69  }
70 }

72 /* m() measures the number of consonant sequences between k0 and j. if c is
73  a consonant sequence and v a vowel sequence, and <..> indicates arbitrary
74  presence,
75 
76  <c><v> gives 0
77  <c>vc<v> gives 1
78  <c>vcvc<v> gives 2
79  <c>vcvcvc<v> gives 3
80  ....
81 */
82 
83 static int m()
84 { int n = 0;
85  int i = k0;
86  while(TRUE)
87  { if (i > j) return n;
88  if (! cons(i)) break; i++;
89  }
90  i++;
91  while(TRUE)
92  { while(TRUE)
93  { if (i > j) return n;
94  if (cons(i)) break;
95  i++;
96  }
97  i++;
98  n++;
99  while(TRUE)
100  { if (i > j) return n;
101  if (! cons(i)) break;
102  i++;
103  }
104  i++;
105  }
106 }

108 /* vowelinstem() is TRUE <=> k0,...j contains a vowel */
109 
110 static int vowelinstem()
111 { int i; for (i = k0; i <= j; i++) if (! cons(i)) return TRUE;
112  return FALSE;
113 }

115 /* doublec(j) is TRUE <=> j,(j-1) contain a double consonant. */
116 
117 static int doublec(int j)
118 { if (j < k0+1) return FALSE;
119  if (b[j] != b[j-1]) return FALSE;
120  return cons(j);
121 }

123 /* cvc(i) is TRUE <=> i-2,i-1,i has the form consonant - vowel - consonant
124  and also if the second c is not w,x or y. this is used when trying to
125  restore an e at the end of a short word. e.g.
126 
127  cav(e), lov(e), hop(e), crim(e), but
128  snow, box, tray.
129 
130 */
131 
132 static int cvc(int i)
133 { if (i < k0+2 || !cons(i) || cons(i-1) || !cons(i-2)) return FALSE;
134  { int ch = b[i];
135  if (ch == 'w' || ch == 'x' || ch == 'y') return FALSE;
136  }
137  return TRUE;
138 }

142 static int ends(char * s)
143 { int length = s[0];
144  if (s[length] != b[k]) return FALSE; /* tiny speed-up */
145  if (length > k-k0+1) return FALSE;
146  if (memcmp(b+k-length+1,s+1,length) != 0) return FALSE;
147  j = k-length;
148  return TRUE;
149 }

151 /* setto(s) sets (j+1),...k to the characters in the string s, readjusting
152  k. */
153 
154 static void setto(char * s)
155 { int length = s[0];
156  memmove(b+j+1,s+1,length);
157  k = j+length;
158 }

186 static void step1ab()
187 { if (b[k] == 's')
188  { if (ends("\04" "sses")) k -= 2; else
189  if (ends("\03" "ies")) setto("\01" "i"); else
190  if (b[k-1] != 's') k--;
191  }
192  if (ends("\03" "eed")) { if (m() > 0) k--; } else
193  if ((ends("\02" "ed") || ends("\03" "ing")) && vowelinstem())
194  { k = j;
195  if (ends("\02" "at")) setto("\03" "ate"); else
196  if (ends("\02" "bl")) setto("\03" "ble"); else
197  if (ends("\02" "iz")) setto("\03" "ize"); else
198  if (doublec(k))
199  { k--;
200  { int ch = b[k];
201  if (ch == 'l' || ch == 's' || ch == 'z') k++;
202  }
203  }
204  else if (m() == 1 && cvc(k)) setto("\01" "e");
205  }
206 }

314 /* In stem(p,i,j), p is a char pointer, and the string to be stemmed is from
315  p[i] to p[j] inclusive. Typically i is zero and j is the offset to the last
316  character of a string, (p[j+1] == '\0'). The stemmer adjusts the
317  characters p[i] ... p[j] and returns the new end-point of the string, k.
318  Stemming never increases word length, so i <= k <= j. To turn the stemmer
319  into a module, declare 'stem' as extern, and delete the remainder of this
320  file.
321 */
