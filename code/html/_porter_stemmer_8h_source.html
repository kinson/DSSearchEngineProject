3 
4  It may be be regarded as cononical, in that it follows the algorithm
5  presented in
6 
7  Porter, 1980, An algorithm for suffix stripping, Program, Vol. 14,
8  no. 3, pp 130-137,
9 
10  only differing from it at the points marked --DEPARTURE-- below.

74 /* Member b is a buffer holding a word to be stemmed. The letters are in
75  b[0], b[1] ... ending at b[z->k]. Member k is readjusted downwards as
76  the stemming progresses. Zero termination is not in fact used in the
77  algorithm.
78 
79  Note that only lower case sequences are stemmed. Forcing to lower case
80  should be done before stem(...) is called.
81 
82 
83  Typical usage is:
84 
85  struct stemmer * z = create_stemmer();
86  char b[] = "pencils";
87  int res = stem(z, b, 6);
88  /- stem the 7 characters of b[0] to b[6]. The result, res,
89  will be 5 (the 's' is removed). -/
90  free_stemmer(z);
91 */

107 /* cons(z, i) is TRUE <=> b[i] is a consonant. ('b' means 'z->b', but here
108  and below we drop 'z->' in comments.
109 */
110 
111 static int cons(struct stemmer * z, int i)
112 { switch (z->b[i])
113  { case 'a': case 'e': case 'i': case 'o': case 'u': return FALSE;
114  case 'y': return (i == 0) ? TRUE : !cons(z, i - 1);
115  default: return TRUE;
116  }
117 }
118 
119 /* m(z) measures the number of consonant sequences between 0 and j. if c is
120  a consonant sequence and v a vowel sequence, and <..> indicates arbitrary
121  presence,
122 
123  <c><v> gives 0
124  <c>vc<v> gives 1
125  <c>vcvc<v> gives 2
126  <c>vcvcvc<v> gives 3
127  ....
128 */
129 
130 static int m(struct stemmer * z)
131 { int n = 0;
132  int i = 0;
133  int j = z->j;
134  while(TRUE)
135  { if (i > j) return n;
136  if (! cons(z, i)) break; i++;
137  }
138  i++;
139  while(TRUE)
140  { while(TRUE)
141  { if (i > j) return n;
142  if (cons(z, i)) break;
143  i++;
144  }
145  i++;
146  n++;
147  while(TRUE)
148  { if (i > j) return n;
149  if (! cons(z, i)) break;
150  i++;
151  }
152  i++;
153  }
154 }

156 /* vowelinstem(z) is TRUE <=> 0,...j contains a vowel */
157 
158 static int vowelinstem(struct stemmer * z)
159 {
160  int j = z->j;
161  int i; for (i = 0; i <= j; i++) if (! cons(z, i)) return TRUE;
162  return FALSE;
163 }

165 /* doublec(z, j) is TRUE <=> j,(j-1) contain a double consonant. */
166 
167 static int doublec(struct stemmer * z, int j)
168 {
169  char * b = z->b;
170  if (j < 1) return FALSE;
171  if (b[j] != b[j - 1]) return FALSE;
172  return cons(z, j);
173 }

175 /* cvc(z, i) is TRUE <=> i-2,i-1,i has the form consonant - vowel - consonant
176  and also if the second c is not w,x or y. this is used when trying to
177  restore an e at the end of a short word. e.g.
178 
179  cav(e), lov(e), hop(e), crim(e), but
180  snow, box, tray.
181 
182 */
183 
184 static int cvc(struct stemmer * z, int i)
185 { if (i < 2 || !cons(z, i) || cons(z, i - 1) || !cons(z, i - 2)) return FALSE;
186  { int ch = z->b[i];
187  if (ch == 'w' || ch == 'x' || ch == 'y') return FALSE;
188  }
189  return TRUE;
190 }

192 /* ends(z, s) is TRUE <=> 0,...k ends with the string s. */
193 
194 static int ends(struct stemmer * z, char * s)
195 { int length = s[0];
196  char * b = z->b;
197  int k = z->k;
198  if (s[length] != b[k]) return FALSE; /* tiny speed-up */
199  if (length > k + 1) return FALSE;
200  if (memcmp(b + k - length + 1, s + 1, length) != 0) return FALSE;
201  z->j = k-length;
202  return TRUE;
203 }

205 /* setto(z, s) sets (j+1),...k to the characters in the string s, readjusting
206  k. */
207 
208 static void setto(struct stemmer * z, char * s)
209 { int length = s[0];
210  int j = z->j;
211  memmove(z->b + j + 1, s + 1, length);
212  z->k = j+length;
213 }

219 /* step1ab(z) gets rid of plurals and -ed or -ing. e.g.
220 
221  caresses -> caress
222  ponies -> poni
223  ties -> ti
224  caress -> caress
225  cats -> cat
226 
227  feed -> feed
228  agreed -> agree
229  disabled -> disable
230

232  mating -> mate
233  meeting -> meet
234  milling -> mill
235  messing -> mess
236 
237  meetings -> meet
238

241 static void step1ab(struct stemmer * z)
242 {
243  char * b = z->b;
244  if (b[z->k] == 's')
245  { if (ends(z, "\04" "sses")) z->k -= 2; else
246  if (ends(z, "\03" "ies")) setto(z, "\01" "i"); else
247  if (b[z->k - 1] != 's') z->k--;
248  }
249  if (ends(z, "\03" "eed")) { if (m(z) > 0) z->k--; } else
250  if ((ends(z, "\02" "ed") || ends(z, "\03" "ing")) && vowelinstem(z))
251  { z->k = z->j;
252  if (ends(z, "\02" "at")) setto(z, "\03" "ate"); else
253  if (ends(z, "\02" "bl")) setto(z, "\03" "ble"); else
254  if (ends(z, "\02" "iz")) setto(z, "\03" "ize"); else
255  if (doublec(z, z->k))
256  { z->k--;
257  { int ch = b[z->k];
258  if (ch == 'l' || ch == 's' || ch == 'z') z->k++;
259  }
260  }
261  else if (m(z) == 1 && cvc(z, z->k)) setto(z, "\01" "e");
262  }
263 }

344  case 'n': if (ends(z, "\03" "ant")) break;
345  if (ends(z, "\05" "ement")) break;
346  if (ends(z, "\04" "ment")) break;
347  if (ends(z, "\03" "ent")) break; return;

366 {

368  z->j = z->k;

376 /* In stem(z, b, k), b is a char pointer, and the string to be stemmed is
377  from b[0] to b[k] inclusive. Possibly b[k+1] == '\0', but it is not
378  important. The stemmer adjusts the characters b[0] ... b[k] and returns
379  the new end-point of the string, k'. Stemming never increases word
380  length, so 0 <= k' <= k.
381 */

384 {
385  if (k <= 1) return k; /*-DEPARTURE-*/
386  z->b = b; z->k = k; /* copy the parameters into z */
387

389  stemming process, although no mention is made of this in the
390  published algorithm. Remove the line to match the published
391  algorithm. */
392 
393  step1ab(z);
394  if (z->k > 0) {
395  step1c(z); step2(z); step3(z); step4(z); step5(z);
